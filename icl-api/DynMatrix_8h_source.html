179  inline void setBounds(unsigned int cols, unsigned int rows, bool holdContent=false, const T &initializer=0) throw (InvalidMatrixDimensionException){
180  if((int)cols == m_cols && (int)rows==m_rows) return;
181  if(!(cols*rows)) throw InvalidMatrixDimensionException("matrix dimensions must be > 0");
182  DynMatrix M(cols,rows,initializer);
183  if(holdContent){
184  unsigned int min_cols = iclMin(cols,(unsigned int)m_cols);
185  unsigned int min_rows = iclMin(rows,(unsigned int)m_rows);
186  for(unsigned int i=0;i<min_cols;++i){
187  for(unsigned int j=0;j<min_rows;++j){
188  M(i,j) = (*this)(i,j);
189  }
190  }
191  }
192  m_cols = cols;
193  m_rows = rows;
194  if(m_data && m_ownData) delete [] m_data;
195  m_data = M.begin();
196  m_ownData = true;
197  M.set_data(0);
198  }
